def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        #   flag  ,   
        swapped = False
        #    ,   i  (  )
        for j in range(0, n-i-1):
            #      ,   
            if arr[j] > arr[j+1]:
                arr[j], arr[j+1] = arr[j+1], arr[j]
                swapped = True
        #       ,   
        if not swapped:
            break
    return arr

import time
import random
import matplotlib.pyplot as plt
#    
def insertion_sort(arr):
    n = len(arr)
    for i in range(1, n):
        key = arr[i]
        j = i - 1
        while j >= 0 and arr[j] > key:
            arr[j + 1] = arr[j]
            j -= 1
        arr[j + 1] = key

#    
def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        swapped = False
        for j in range(0, n - i - 1):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
                swapped = True
        if not swapped:
            break

#     
def measure_time(sort_function, arr):
    start_time = time.time()
    sort_function(arr)
    return time.time() - start_time

#   n=[5, 10, 50, 100, 500, 1000, 2000, 3000, 4000, 5000, 10000, 20000, 50000, 100000]
sizes = [5, 10, 50, 100, 500, 1000, 2000, 3000, 4000, 5000]

#    
times_insert = []
times_bubble = []

#      
for n in sizes:
    arr_insert = random.sample(range(1, n*10), n)
    arr_bubble = arr_insert.copy()
    
    #      
    time_insert = measure_time(insertion_sort, arr_insert)
    times_insert.append(time_insert)
    
    #      
    time_bubble = measure_time(bubble_sort, arr_bubble)
    times_bubble.append(time_bubble)

#  
plt.figure(figsize=(10, 6))
plt.plot(sizes, times_insert, label='Insertion Sort', marker='o', color='b')
plt.plot(sizes, times_bubble, label='Bubble Sort', marker='s', color='r')
plt.xlabel('  (n)')
plt.ylabel('  ()')
plt.title('   ')
plt.legend()
plt.grid(True)
plt.show()
